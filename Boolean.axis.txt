package axis.lang



//包级别类 <<<<<<<<<<<<<<<<<<<<<<

Any Boolean { 

    final String TRUE : “true”
    final String FALSE : “false”

    @Override
    String toString() {
      return this ? TRUE ; FALSE
    }

    Boolean OF(String s) {
      s = s ?? “”
      return Boolean{
        value : s = TRUE ? true ; false 
      }
    }
}

//包级别类 >>>>>>>>>>>>>>>>>>>>>>
