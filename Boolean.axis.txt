package axis.lang



//包级别类 <<<<<<<<<<<<<<<<<<<<<<

Boolean : Any { 

    final TRUE : String = “true”
    final FALSE : String = “false”

    @Override
    toString() : String {
      return this ? TRUE : FALSE
    }

    OF(s : String) {
      s = s ?? “”
      return Boolean{
        value : s == TRUE ? true : false 
      }
    }
}

//包级别类 >>>>>>>>>>>>>>>>>>>>>>
