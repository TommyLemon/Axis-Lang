package axis.lang



//包级别类 <<<<<<<<<<<<<<<<<<<<<<

Any Bool { 

    final Char TRUE : “true”
    final Char FALSE : “false”

    @Override
    Char toChar() {
      return this ? TRUE ; FALSE
    }

    Bool OF(Char s) {
      s = s ?? “”
      return Bool{
        value : s = TRUE ? true ; false 
      }
    }
}

//包级别类 >>>>>>>>>>>>>>>>>>>>>>
